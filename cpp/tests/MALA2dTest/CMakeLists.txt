project (UQTk)

INSTALL(FILES ${copy_FILES} DESTINATION cpp/tests/MALA2dTest/)

add_executable (MALA2dTest main.cpp)

target_link_libraries (MALA2dTest uqtk  )

target_link_libraries (MALA2dTest depdsfmt )
target_link_libraries (MALA2dTest sundials_cvode)
target_link_libraries (MALA2dTest sundials_nvecserial)
target_link_libraries (MALA2dTest sundials_sunlinsoldense)
target_link_libraries (MALA2dTest sundials_sunmatrixdense)
target_link_libraries (MALA2dTest depslatec)
target_link_libraries (MALA2dTest m lapack ${LAPACK_LIBRARIES})
target_link_libraries (MALA2dTest m blas ${BLAS_LIBRARIES})
target_link_libraries (MALA2dTest deplbfgs  )

# Link fortran libraries
if ("${CMAKE_CXX_COMPILER_ID}" STREQUAL "GNU")
  # using GCC
  target_link_libraries (MALA2dTest gfortran expat stdc++)
elseif ("${CMAKE_CXX_COMPILER_ID}" STREQUAL "Intel")
  # using Intel
  if ("${IntelLibPath}" STREQUAL "")
    target_link_libraries (MALA2dTest ifcore ifport)
  else()
    target_link_libraries (MALA2dTest ${IntelLibPath}/libifcore.a)
    target_link_libraries (MALA2dTest ${IntelLibPath}/libifport.a)
  endif()
elseif (CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  # using Clang
  if ("${ClangLibPath}" STREQUAL "")
    target_link_libraries (MALA2dTest gfortran stdc++)
  else()
    target_link_libraries (MALA2dTest ${ClangLibPath}/libgfortran.dylib  ${ClangLibPath}/libstdc++.dylib)
  endif()
endif()

include_directories(../../lib/pce    )
include_directories(../../lib/array  )
include_directories(../../lib/include)
include_directories(../../lib/quad   )
include_directories(../../lib/tools  )
include_directories(../../lib/mcmc  )

include_directories(../../../dep/dsfmt)
include_directories(../../../dep/slatec)
include_directories(../../../dep/lbfgs)

INSTALL(TARGETS MALA2dTest DESTINATION bin/tests/)

add_test(MALA2dTest MALA2dTest)
